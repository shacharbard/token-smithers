GREEN — src/token_sieve/hooks/posttooluse_metrics.py:
  ```python
  def main() -> int:
      try:
          payload = json.load(sys.stdin)
          tool_name = payload.get("tool_name")
          tool_output = payload.get("tool_output", "")
          duration_ms = int(payload.get("duration_ms", 0))
          session_id = os.environ.get("CLAUDE_SESSION_ID", "default")
          if not tool_name:
              return 0
          store = SQLStore.default()
          store.record_post_tool_use_metric(
              tool_name=tool_name,
              output_size_bytes=len(tool_output.encode("utf-8")),
              duration_ms=duration_ms,
              session_id=session_id,
          )
      except Exception:
          pass  # fail open
      return 0
  ```
