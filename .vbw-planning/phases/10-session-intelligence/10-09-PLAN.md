  ```python
  def connected_components(graph: ImportsGraph) -> dict[str, int]:
      # Build undirected adjacency
      adj: dict[str, set[str]] = defaultdict(set)
      for src, dsts in graph.forward.items():
          for dst in dsts:
              adj[src].add(dst)
              adj[dst].add(src)
      # BFS per unvisited node, assign in lexicographic order
      visited: dict[str, int] = {}
      cid = 0
      for root in sorted(adj.keys()):
          if root in visited: continue
          queue = deque([root])
          while queue:
              n = queue.popleft()
              if n in visited: continue
              visited[n] = cid
              queue.extend(adj[n])
          cid += 1
      return visited
  ```

GREEN — src/token_sieve/session/cluster_index.py:
  ```python
  class ClusterIndex:
      def __init__(self, tracker: SessionContextTracker, component_map: dict[str, int]):
          self._tracker = tracker
          self._map = component_map
          self._visited: set[int] = set()
      def add(self, content: str, key: str, file_path: str | None) -> None:
          self._tracker.add(content, key)
          if file_path and file_path in self._map:
              self._visited.add(self._map[file_path])
      def cluster_for(self, file_path: str) -> int | None:
          return self._map.get(file_path)
      def cluster_visited(self, cluster_id: int) -> bool:
          return cluster_id in self._visited
  ```
