  ```python
  def run_auto_tune(store: SQLStore, config, seed: int = 0) -> dict[str, float]:
      try:
          stats = store.adapter_effectiveness_stats()
      except Exception:
          logger.warning("auto-tune: failed to read store, skipping")
          return {}
      adjustments = {}
      for adapter, (savings_ratio, sample_count) in stats.items():
          if sample_count < 20: continue
          target = _target_multiplier(savings_ratio)
          current = store.get_adapter_budget_multiplier(adapter) or 1.0
          capped = _cap_adjustment(current, target, max_delta=0.2)
          if capped != current:
              adjustments[adapter] = capped
              store.set_adapter_budget_multiplier(adapter, capped)
      return adjustments
  ```
