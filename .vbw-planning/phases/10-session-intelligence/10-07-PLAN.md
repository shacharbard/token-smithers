  ```python
  class JCodeMunchOracle:
      def __init__(self, mcp_client, path_for_symbol: Callable[[str], str]):
          self._client = mcp_client
          self._path_for_symbol = path_for_symbol
      def blast_radius(self, changed_paths: set[str]) -> set[str]:
          result = set(changed_paths)
          for path in changed_paths:
              symbols = self._client.call_tool("mcp__jcodemunch__get_blast_radius", {"file_path": path})
              for sym in symbols:
                  p = self._path_for_symbol(sym)
                  if p: result.add(p)
          return result
  ```
