GREEN — src/token_sieve/session/context_tracker.py:
  ```python
  class SessionContextTracker:
      def __init__(self, num_perm: int = 128, threshold: float = 0.7) -> None:
          self._num_perm = num_perm
          self._threshold = threshold
          self._lsh = MinHashLSH(threshold=threshold, num_perm=num_perm)
          self._signatures: dict[str, MinHash] = {}
      def _signature(self, content: str) -> MinHash:
          mh = MinHash(num_perm=self._num_perm)
          for token in content.split():
              mh.update(token.encode("utf-8"))
          return mh
      def add(self, content: str, key: str) -> None:
          if key in self._signatures:
              self._lsh.remove(key)
          sig = self._signature(content)
          self._signatures[key] = sig
          self._lsh.insert(key, sig)
      def query_similar(self, content: str, threshold: float | None = None) -> list[str]:
          sig = self._signature(content)
          return list(self._lsh.query(sig))
  ```
Commit GREEN.
